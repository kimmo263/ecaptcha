集成代码

<script src="/ecaptcha/sdk/ecaptcha.min.js"></script>
<script>
  // 初始化
  ECaptcha.init({ server: window.location.origin });
  
  // 使用
  async function sendCode() {
    const token = await ECaptcha.smart();
    // 发送验证码时带上 token
    fetch('/user/v1/auth/send-code', {
      method: 'POST',
      body: JSON.stringify({
        phone: '13800138000',
        captcha_token: token
      })
    });
  }
</script>
